## Array.prototype.concat()

function myConcat(...args) {
  const res = []
  const argsLen = args.length

  for (let i = 0; i < argsLen; i++) {
    if (isArray(args[i])) {
      const arr = args[i]
      const arrLen = arr.length

      for (let j = 0; j < arrLen; j++) {
        res.push(arr[j])
      }
    }
    else {
      res.push(args[i])
    }
  }

  return res
}

function isArray(obj) {
  return Object.prototype.toString.call(obj) === '[object Array]'
}

// test

const arr2 = [4, 5, 6]
console.log(myConcat(0, arr1, arr2)) // [0, 1, 2, 3, 4, 1, 2]

## Array.prototype.copyWithin()

```js
function copyWithin(arr, target, start, end) {
  const len = arr.length
  const copy = []
  const copyLen = end - start

  target = Math.floor(target)
  start = Math.floor(start)
  end = Math.floor(end)

  if (target >= len || start >= len || end < start) {
    return arr
  }

  if (target < -len) {
    target = 0
  }
  else if (target < 0) {
    target = target + len
  }

  if (start < -len) {
    start = 0
  }
  else if (start < 0) {
    start = start + len
  }

  if (end < -len) {
    end = 0
  }
  else if (end < 0) {
    end = end + len
  }
  else if (end > len) {
    end = len
  }

  for (let i = start; i < end; i++) {
    copy[copy.length] = arr[i]
  }

  for (let i = 0; i < copyLen; i++) {
    arr[target++] = copy[i]
    if (target === len) {
      break
    }
  }

  return arr
}

const arr = ['a', 'b', 'c', 'd', 'e']
console.log(copyWithin(arr, 3, 2, 5)) // ['a', 'b', 'c', 'c', 'd']
```

function myFlat(arr, depth = 1) {
  const len = arr.length
  const res = []

  for (let i = 0; i < len; i++) {
    const item = arr[i]

    if (isArray(item) && depth > 0) {
      const sub = myFlat(item, depth - 1)
      for (let j = 0; j < sub.length; j++) {
        res.push(sub[j])
      }
    }
    else {
      res.push(item)
    }
  }

  return res
}

## Array.prototype.pop()

function myPop(arr) {
  const len = arr.length
  const lastIndex = len > 0 ? len - 1 : -1
  if (lastIndex >= 0) {
    const last = arr[lastIndex]
    arr.length -= 1
    return last
  }
}

function myReduce(arr, cb, initialValue) {
  const len = arr.length
  let i = initialValue === undefined ? 1 : 0
  initialValue = initialValue === undefined ? arr[0] : initialValue
  let accumulator = initialValue

  for (; i < len; i++) {
    accumulator = cb.call(undefined, accumulator, arr[i], arr)
  }
  return accumulator
}

function myReverse(arr) {
  const len = arr.length
  const res = []

  for (let i = len - 1; i >= 0; i--) {
    res[res.length] = arr[i]

## Array.prototype.splice()

```js
function mySplice(arr, start, deleteCount, ...items) {
  const len = arr.length

  if (arguments.length === 1) {
    start = len
  }

  start = Math.floor(start)
  if (start < -len) {
    start = 0
  }
  if (start < 0) {
    start = start + len
  }

  deleteCount = Math.floor(deleteCount)
  const tail = []
  for (let i = start + deleteCount; i < len; i++) {
    tail[tail.length] = arr[i]
  }

  arr.length = start
  const itemsCount = items.length
  for (let i = 0; i < itemsCount; i++) {
    arr[arr.length] = items[i]
  }

  const tailLen = tail.length
  for (let i = 0; i < tailLen; i++) {
    arr[arr.length] = tail[i]
  }

  return arr
}

// test
const arr = [1, 2, 3, 4, 5]
console.log(mySplice(arr, 1, 1, 3)) // [1, 3, 3, 4, 1]
```

## Array.prototype.unshift()

```js
function myUnshift(arr, ...items) {
  const itemsLen = items.length

  if (itemsLen === 0) {
    return arr
  }

  const len = arr.length
  const tmp = []

  for (let i = 0; i < len; i++) {
    tmp[i] = arr[i]
  }

  arr.length = itemsLen
  for (let i = 0; i < itemsLen; i++) {
    arr[i] = items[i]
  }

  for (let i = 0; i < len; i++) {
    arr[arr.length] = tmp[i]
  }

  return arr
}

// test
const arr = [1, 2, 3, 4, 5]
console.log(myUnshift(arr, 6, 7)) // [2, 7, 1, 3, 3, 4, 1]
```

## Array.prototype.with()

function myWith(arr, index, value) {
  const len = arr.length
  const res = []

  for (let i = 0; i < len; i++) {
    res[i] = arr[i]
  }
  if (index > len || index < -len) {
    throw new RangeError(`Invalid index : ${index}`)
  }

  if (index < 0) {
    index = index + len
  }

  res[index] = value

  return res
}
